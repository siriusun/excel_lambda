/*
No: 1
Name: myFunc.isin
Description: 判断目标是否在列表中，返回1，0
Category: Lookup
Enabled: Y
*/
myFunc.isin = LAMBDA(value, array,
    N(
        ISNUMBER(
            XMATCH(
                "t" & value,
                "t" & TRIMRANGE(array)
            )
        )
    )
);

/*
No: 2
Name: myFunc.getHC
Description: 获取流水线大号, 非流水线第二参数0返回空，1返回原号码，兼容H开头
Category: Serial
Enabled: Y
*/
myFunc.getHC = LAMBDA(father_sn, [return_0Null_1Origin],
    IF(
        REGEXTEST(
            father_sn,
            "^H?167[0-9]{8}.[0-9]{3}$"
        ),
        IF(
            ISNUMBER(FIND("-", father_sn)),
            "H" & RIGHT(father_sn, 3),
            "H0" & RIGHT(father_sn, 2)
        ),
        IF(
            OR(
                return_0Null_1Origin = 0,
                ISOMITTED(
                    return_0Null_1Origin
                )
            ),
            "",
            father_sn
        )
    )
);

/*
No: 5
Name: myFunc.getWeek
Description: 获取目标日期的周一与周日的格式文本 yyyy/mm/dd - mm/dd
Category: Date
Enabled: Y
*/
myFunc.getWeek = LAMBDA(date,
    LET(
        week_start, date -
            WEEKDAY(date, 2),

    )
);

/*
No: 8
Name: myFunc.replaceMap
Description: 对一个文本进行对照翻译，列表1 => 列表2映射
Category: Text
Enabled: Y
*/
myFunc.replaceMap = LAMBDA(text, oldArray, newArray,
    LET(
        oldCount, COUNTA(oldArray),
        seq, SEQUENCE(oldCount),
        REDUCE(
            text,
            seq,
            LAMBDA(result, idx,
                SUBSTITUTE(
                    result,
                    INDEX(oldArray, idx),
                    INDEX(newArray, idx)
                )
            )
        )
    )
);

/*

Name: myFunc.flipArray

Category: Array
Enabled: Y

myFunc.flipArray = LAMBDA(one_dimension_arr,
    LET(
        arr, one_dimension_arr,
        order, MAKEARRAY(

myFunc.flipMatrix = LAMBDA(matrix, [dft_lr_1lr_2ud],

        rlt
    )
);

/*
No: 15
Name: myFunc.logTest
Description: 检验acltop log文件名，SA日志CHINA开头，DBExtract要DBX开头

myFunc.logTest = LAMBDA(toplog_filename,
    LET(

                REGEXTEST(
                    TEXTAFTER(
                        toplog_filename,
                        "DBX_"

                    "^[0-9]{8}J?_[0-9]{4}-[0-9]{2}-[0-9]{2}_[0-9]{2}-[0-9]{2}-[0-9]{2}.+$"
                )
            ),
            -1
        )
    )
);

/*
No: 18
Name: myFunc.strHash
Description: 获取字符串的伪hash值，默认9位
Category: Text
Enabled: Y
*/
myFunc.strHash = LAMBDA(text, [length_default9],
    LET(
        hashLength, IF(
            ISOMITTED(length_default9),
            9,
            length_default9
        ),
        iterator, SEQUENCE(LEN(text)),
        rollingHash, REDUCE(
            0,
            iterator,
            LAMBDA(acc, i,
                MOD(
                    MOD(
                        acc * 31,
                        POWER(
                            10,
                            hashLength
                        )
                    ) +
                        UNICODE(
                            MID(text, i, 1)
                        ),
                    POWER(10, hashLength)
                )
            )
        ),
        numericHash, MOD(
            rollingHash,
            POWER(10, hashLength)
        ),
        numericString, TEXT(
            numericHash,
            REPT("0", hashLength)
        ),
        "H" & numericString
    )
);

/*
No: 20
Name: myFunc.NormalizeValue
Description: 均一化公式，1正向0反向
Category: Math
Enabled: Y
*/
myFunc.NormalizeValue = LAMBDA(arr, target, [Default_1_1Pos0Neg],
    LET(
        dir, IF(
            ISOMITTED(Default_1_1Pos0Neg),
            1,
            Default_1_1Pos0Neg
        ),
        minVal, MIN(arr),
        maxVal, MAX(arr),
        normalizedTarget, IF(
            minVal = maxVal,
            NA(),
            SWITCH(
                TRUE(),
                ISBLANK(target), "",
                dir = 0, (
                    maxVal - target
                ) / (maxVal - minVal),
                dir = 1, (
                    target - minVal
                ) / (maxVal - minVal),
                NA()
            )
        ),
        normalizedTarget
    )
);

/*
No: 21
Name: myFunc.restr
Description: 把目标字符数组合并成正则字符串，1精确 0模糊
Category: Regex
Enabled: Y
*/
myFunc.restr = LAMBDA(arr, [mode],
    LET(
        m, IF(ISOMITTED(mode), 1, mode),
        a, TOCOL(arr, 1),
        b, FILTER(TRIM(a), TRIM(a) <> ""),
        c, IF(m = 0, ".*" & b, b),
        "(" & TEXTJOIN("|", TRUE, c) & ")"
    )
);

/*
No: 22
Name: myFunc.RegexFilter
Description: 正则筛选一个矩阵，支持ANY/ALL/CONCAT模式
Category: Regex
Enabled: Y
*/
myFunc.RegexFilter = LAMBDA(
    arr,
    pattern,
    [ignore_case],
    [empty_fallback],
    [match_mode],
    [concat_delim],
    LET(
        a, arr,
        ic, IF(
            ISOMITTED(ignore_case),
            1,
            ignore_case
        ),
        fb, IF(
            ISOMITTED(empty_fallback),
            "null",
            empty_fallback
        ),
        mm, IF(
            ISOMITTED(match_mode),
            "ANY",
            UPPER(match_mode)
        ),
        delim, IF(
            ISOMITTED(concat_delim),
            " ",
            concat_delim
        ),
        include, IF(
            COLUMNS(a) = 1,
            REGEXTEST(a & "", pattern, ic),
            BYROW(
                a,
                LAMBDA(r,
                    SWITCH(
                        mm,
                        "ANY", OR(
                            BYCOL(
                                r,
                                LAMBDA(c,
                                    REGEXTEST(
                                        c &
                                            "",
                                        pattern,
                                        ic
                                    )
                                )
                            )
                        ),
                        "ALL", AND(
                            BYCOL(
                                r,
                                LAMBDA(c,
                                    REGEXTEST(
                                        c &
                                            "",
                                        pattern,
                                        ic
                                    )
                                )
                            )
                        ),
                        "CONCAT", REGEXTEST(
                            TEXTJOIN(
                                delim,
                                TRUE,
                                r
                            ),
                            pattern,
                            ic
                        ),
                        OR(
                            BYCOL(
                                r,
                                LAMBDA(c,
                                    REGEXTEST(
                                        c &
                                            "",
                                        pattern,
                                        ic
                                    )
                                )
                            )
                        )
                    )
                )
            )
        ),
        IFERROR(FILTER(a, include), fb)
    )
);

/*
No: 23
Name: myFunc.Calendar
Description: 接收开始日期与完成日期两个参数，生成完整日历表
Category: Calendar
Enabled: Y
*/
myFunc.Calendar = LAMBDA(startDate, endDate,
    IF(
        startDate > endDate,
        "开始日期不能晚于结束日期",
        LET(
            dates, SEQUENCE(endDate - startDate + 1, , startDate, 1),
            VSTACK(
                {
                    "Date",
                    "Year",
                    "Quarter",
                    "yQuarter",
                    "Month",
                    "YearMonth",
                    "yMonth",
                    "Day",
                    "WeekdayNo",
                    "WeekdayName",
                    "WeekStart",
                    "WeekEnd",
                    "IsWeekend",
                    "MonthStart",
                    "MonthEnd"
                },
                HSTACK(
                    dates,
                    YEAR(dates),
                    "Q" & ROUNDUP(MONTH(dates) / 3, 0),
                    YEAR(dates) & "Q" & ROUNDUP(MONTH(dates) / 3, 0),
                    MONTH(dates),
                    TEXT(dates, "yyyy-mm"),
                    TEXT(dates, "yymm"),
                    DAY(dates),
                    WEEKDAY(dates, 2),
                    TEXT(dates, "aaa"),
                    dates - WEEKDAY(dates, 2) + 1,
                    dates - WEEKDAY(dates, 2) + 7,
                    IF(WEEKDAY(dates, 2) > 5, "Yes", "No"),
                    EOMONTH(dates, -1) + 1,
                    EOMONTH(dates, 0)
                )
            )
        )
    )
);
